90 template <typename T>
91 inline auto
92 hash(const std::string &suffix = "") -> std::string
93 {
94 using underlying_type =
95 std::remove_cv_t<std::remove_reference_t<T>>;
96 using stored_type = std::shared_ptr<underlying_type>;
97
98 const auto alias_it =
99 detail::type_aliases().find(typeid(underlying_type).hash_code());
100
101 const auto resolved_underlying = entt::resolve<underlying_type>();
102 std::string base_identifier =
103 (alias_it != detail::type_aliases().end()) ?
104 alias_it->second :
105 (resolved_underlying ?
106 detail::type_identifier(resolved_underlying) :
107 std::string(entt::type_id<underlying_type>().name()));
108
109 const char *base_name_ptr = detail::store_identifier(base_identifier);
110 const auto type_id = entt::hashed_string{base_name_ptr}.value();
111 entt::meta_factory<stored_type>().type(type_id, base_name_ptr);
112
113 if (suffix.empty())
114 return base_identifier;
115
116 // For function/method types, prefer the supplied suffix (usually a name)
117 // over the full signature to keep identifiers readable and stable.
118 if (base_identifier.find('(') != std::string::npos)
119 {
120 detail::store_identifier(suffix);
121 return suffix;
122 }
123
124 const auto identifier = base_identifier + suffix;
125 detail::store_identifier(identifier);
126 return identifier;
127 }

536 template <typename T>
537 static auto
538 register_json_header(const std::string &suffix = "")
539 -> NodeObjectInitializer &
540 {
541 auto hash_str = coral::hash<T>(suffix);
542 if (initializers.find(hash_str) != initializers.end())
543 // Reset the initializer
544 initializers[hash_str] = {};
545
546 auto &initializer = initializers[hash_str];
547 initializer.type_name = boost::core::type_name<T>();
548 initializer.json_serializer["type"] = hash_str;

550 initializer.json_serializer["inputs"] = json::array();
551 initializer.json_serializer["outputs"] = json::array();
552 return initializer;

559 template <typename T, typename... Args>
560 static auto
561 register_json_header(const std::vector<std::string> &arg_names,
562 const std::string &suffix = "")
563 -> NodeObjectInitializer &
564 {
565 auto &initializer = register_json_header<T>(suffix);
566
567 // Now take care of the arguments.
568 std::vector<std::shared_ptr<NodeObject>> args = {std::make_shared<
569 NodeObject>(
570 std::shared_ptr<std::remove_cv_t<std::remove_reference_t<Args>>>())...};
571
572 std::vector<ConnectionType> arg_connection_types = {
573 connection_type<Args>()...};
574
575 if (args.size() != arg_names.size())
576 throw std::runtime_error("Wrong number of arguments.");
577
578 for (unsigned int i = 0; i < args.size(); ++i)
579 {
580 initializer.json_serializer["arguments"][i]["name"] = arg_names[i];
581 initializer.json_serializer["arguments"][i]["type"] = args[i]->hash();
582 initializer.json_serializer["arguments"][i]["connection_type"] =
583 magic_enum::enum_name(arg_connection_types[i]);
584 if ((arg_connection_types[i] & ConnectionType::input) !=
585 ConnectionType::none)
586 initializer.json_serializer["inputs"].push_back(i);
587 if ((arg_connection_types[i] & ConnectionType::output) !=
588 ConnectionType::none)
589 initializer.json_serializer["outputs"].push_back(i);
590 }
591 return initializer;
592 }

600 template <typename T>
601 static auto
602 register_elementary_type() -> NodeObjectInitializer &
603 {
604 auto &initializer = register_json_header<T>();
605 initializer.json_serializer["node_type"] = "elementary_constructor";
606 initializer.node_type = NodeType::elementary_constructor;
607
608 // Special handling for std::string
609 if constexpr (std::is_same_v<T, std::string>)
610 {
611 initializer.json_serializer["value"] = ""; // Empty string default
612 }
613 else
614 {
615 initializer.json_serializer["value"] = json(T()).dump();
616 }
617
618 initializer.json_serializer["outputs"].push_back(-1);
619
620 // Add value parser
621 initializer.parse_string =
622 [](const std::string &value) -> std::shared_ptr<entt::meta_any> {
623 auto t = std::make_shared<T>();
624
625 if constexpr (std::is_same_v<T, std::string>)
626 {
627 *t = value;
628 }
629 else
630 {
631 // For non-string types, use standard JSON parsing
632 *t = json::parse(value).get<T>();
633 }
634
635 return std::make_shared<entt::meta_any>(t);
636 };
637
638 // Add to the initializer the emtpy executor.
639 initializer.executor =
640 [](const std::vector<std::shared_ptr<NodeObject>> &args)
641 -> std::shared_ptr<entt::meta_any> {
642 if (args.size() != 0)
643 throw std::runtime_error("Wrong number of arguments.");
644 return std::make_shared<entt::meta_any>(std::make_shared<T>());
645 };
646
647 // Add value parser
648 initializer.to_string =
649 [](const std::shared_ptr<entt::meta_any> &value) -> std::string {
650 const auto ptr = value->template try_cast<std::shared_ptr<T>>();
651 if (ptr == nullptr)
652 throw std::runtime_error(
653 "Could not cast meta_any to requested shared_ptr type.");
654 const T &t = **ptr;
655
656 if constexpr (std::is_same_v<T, std::string>)
657 {
658 return t;
659 }
660 else
661 {
662 // For non-string types, use standard JSON serialization
663 return json(t).dump();
664 }
665 };
666 return initializer;
667 }

1149 auto
1150 get_shared() -> std::shared_ptr<T>
1151 {
1152 if (!ready())
1153 throw std::runtime_error("Object is not ready.");
1154 using type = std::remove_cv_t<std::remove_reference_t<T>>;
1155 std::shared_ptr<type> ptr;
1156
1157 if (hash() != coral::hash<type>())
1158 {
1159 auto &j = initializer.json_serializer;
1160 if (!(j.contains("base") &&
1161 (coral::hash<type>() == j.at("base").get<std::string>())))
1162 throw std::runtime_error("Cannot cast object of type " +
1163 type_name() + " to object of type " +
1164 boost::core::type_name<type>() + ".");
1165 auto new_object = initializer.to_base(object);
1166 if (!(new_object && *new_object))
1167 throw std::runtime_error("New object does not have value.");
1168 if (!(coral::hash(new_object) == j.at("base").get<std::string>()))
1169 throw std::runtime_error(
1170 "New object does not have the right hash.");
1171 const auto cast_ptr =
1172 new_object->template try_cast<std::shared_ptr<type>>();
1173 if (cast_ptr == nullptr)
1174 throw std::runtime_error("Could not cast converted object to " +
1175 std::string(boost::core::type_name<type>()));
1176 ptr = *cast_ptr;
1177 }
1178 else
1179 {
1180 const auto cast_ptr =
1181 object->template try_cast<std::shared_ptr<type>>();
1182 if (cast_ptr == nullptr)
1183 throw std::runtime_error(
1184 "Could not cast object to shared pointer of type " +
1185 std::string(boost::core::type_name<type>()) + " from object of type " +
1186 boost::core::demangle([&]() {
1187 const char *named = object->type().name();
1188 if (named != nullptr && named[0] != '\0')
1189 return named;
1190 const auto info_name = object->type().info().name();
1191 static thread_local std::string info_name_str;
1192 info_name_str = std::string(info_name);

1574

1576 // from the base type
1577 const auto &base_hash =
1578 initializer.json_serializer["arguments"][input_indices[i]].value(
1579 "base", "");

1581 (expected_hash == input_hash) || (base_hash == input_hash);
1582
1583 if (!is_valid)
1584 throw std::runtime_error(
1585 "The hash type of input " + std::to_string(i) + " (" +
1586 input_hash + ") does not match the expected hash (" +
1587 expected_hash + ") or its base type (" + base_hash + ".");

1636 inline void
1637 from_json(const json &j, NodeObjectPtr &obj)
1638 {
1639 if (!j.contains("type"))
1640 throw std::runtime_error(
1641 "The json does not contain a type entry. Bailing out.");
1642 obj = make_node(j.at("type").get<std::string>());
1643 // Make sure the hash matches the expected value
1644 if (j["type"] != obj->hash())
1645 throw std::runtime_error(
1646 "The type does not match the expected value: expected " +
1647 j["type"].dump() + ", got " + obj->hash());
1648
1649 const auto &reg = obj->get_info();
1650
1651 // If the input JSON contains fields that also exist in the registry
1652 // description, they must match exactly (except for "value", which is
1653 // allowed to differ when provided).
1654 for (const auto &[key, val] : j.items())
1655 {
1656 if (key == "value")
1657 continue;
1658 if (reg.contains(key) && reg.at(key) != val)
1659 throw std::runtime_error(
1660 "The json object field '" + key +
1661 "' does not match the registered value: expected " +
1662 reg.at(key).dump() + ", got " + val.dump());
1663 }
1664
1665 const auto node_type = reg.value("node_type", "");
1666
1667 if (node_type == "elementary_constructor" ||
1668 node_type == "empty_constructor")
1669 (*obj)();
1670 if (j.contains("value"))
1671 {
1672 obj->parse_string(j.at("value").get<std::string>());
1673 if (obj->ready())
1674 {
1675 // Refresh serialized value to match the parsed object.
1676 (void)obj->get_info();
1677 }
1678 }
