134 template <typename T>
135 inline auto
136 hash(const std::string &suffix = "") -> std::string
137 {
138 using underlying_type = std::remove_cv_t<std::remove_reference_t<T>>;
139 using stored_type = std::shared_ptr<underlying_type>;
140
141 std::string alias;
142 {
143 const std::lock_guard<std::mutex> lock(type_aliases_mutex());
144 const auto it =
145 type_aliases().find(typeid(underlying_type).hash_code());
146 if (it != type_aliases().end())
147 alias = it->second;
148 }
149
150 const auto resolved_underlying = entt::resolve<underlying_type>();
151 std::string base_identifier =
152 !alias.empty() ?
153 alias :
154 (resolved_underlying ?
155 type_identifier(resolved_underlying) :
156 std::string(entt::type_id<underlying_type>().name()));
157
158 const char *base_name_ptr = store_identifier(base_identifier);
159 const char *type_name_ptr = base_name_ptr;
160 if (!suffix.empty() && base_identifier.find('(') != std::string::npos)
161 type_name_ptr = store_identifier(suffix);
162 const auto type_id = entt::hashed_string{type_name_ptr}.value();
163 entt::meta_factory<stored_type>().type(type_id, type_name_ptr);
164
165 if (suffix.empty())
166 return base_identifier;
167
168 // For function/method types, prefer the supplied suffix (usually a name)
169 // over the full signature to keep identifiers readable and stable.
170 if (base_identifier.find('(') != std::string::npos)
171 {
172 store_identifier(suffix);
173 return suffix;
174 }
175
176 const auto identifier = base_identifier + suffix;
177 store_identifier(identifier);
178 return identifier;
179 }

620 template <typename T>
621 static auto
622 register_elementary_type() -> detail::NodeObjectInitializer &
623 {
624 auto &initializer = register_json_header<T>();
625 initializer.json_serializer["node_type"] = "elementary_constructor";
626 initializer.node_type = NodeType::elementary_constructor;
627
628 // Special handling for std::string
629 if constexpr (std::is_same_v<T, std::string>)
630 {
631 initializer.json_serializer["value"] = ""; // Empty string default
632 }
633 else
634 {
635 initializer.json_serializer["value"] = json(T()).dump();
636 }
637
638 initializer.json_serializer["outputs"].push_back(-1);
639
640 // Add value parser
641 initializer.parse_string =
642 [](const std::string &value) -> std::shared_ptr<entt::meta_any> {
643 auto t = std::make_shared<T>();
644
645 if constexpr (std::is_same_v<T, std::string>)
646 {
647 *t = value;
648 }
649 else
650 {
651 // For non-string types, use standard JSON parsing
652 *t = json::parse(value).get<T>();
653 }
654
655 return std::make_shared<entt::meta_any>(t);
656 };
657
658 // Add to the initializer the emtpy executor.
659 initializer.executor = [](const NodeObjectPtr &,
660 const std::vector<NodeObjectPtr> &args)
661 -> std::shared_ptr<entt::meta_any> {
662 if (args.size() != 0)
663 throw std::runtime_error("Wrong number of arguments.");
664 return std::make_shared<entt::meta_any>(std::make_shared<T>());
665 };
666
667 // Add value parser
668 initializer.to_string =
669 [](const std::shared_ptr<entt::meta_any> &value) -> std::string {
670 const auto ptr = value->template try_cast<std::shared_ptr<T>>();
671 if (ptr == nullptr)
672 throw std::runtime_error(
673 "Could not cast meta_any to requested shared_ptr type.");
674 const T &t = **ptr;
675
676 if constexpr (std::is_same_v<T, std::string>)
677 {
678 return t;
679 }
680 else
681 {
682 // For non-string types, use standard JSON serialization
683 return json(t).dump();
684 }
685 };
686 return initializer;
687 }

814 template <typename T, typename ReturnType, typename... Args>
815 static void
816 register_method(MethodPtr<T, ReturnType, Args...> ptr,
817 std::vector<std::string> arg_names)
818 {
819 using ThisMethod = MethodPtr<T, ReturnType, Args...>;
820 constexpr bool return_is_void = std::is_same_v<ReturnType, void>;
821 auto method_name = arg_names[0];
822 arg_names.erase(arg_names.begin());
823
824 if constexpr (return_is_void)
825 {
826 auto &initializer =
827 register_json_header<ThisMethod, T &, Args...>(arg_names,
828 method_name);
829 initializer.json_serializer["node_type"] = "void_method";
830 initializer.node_type = NodeType::void_method;
831
832 // Add the method to the initializer
833 initializer.executor =
834 [ptr, &initializer, method_name](const NodeObjectPtr &,
835 std::vector<NodeObjectPtr> args)
836 -> std::shared_ptr<entt::meta_any> {
837 if (args.size() != 1 + sizeof...(Args))
838 throw std::runtime_error("Wrong number of arguments.");
839 auto &obj = args[0]->get<T>();
840 args.erase(args.begin()); // remove the first element
841
842 auto tuple = detail::cast_args<Args...>(args);
843 slog_debug("%s: %s",
844 method_name.c_str(),
845 initializer.type_name.c_str());
846 std::apply(
847 [&obj, ptr](auto &&...unpackedArgs) {
848 (obj.*ptr)(*unpackedArgs...);
849 },
850 tuple);
851 return std::make_shared<entt::meta_any>(
852 std::make_shared<ThisMethod>(ptr));
853 };
854 }
855 else
856 {
857 auto &initializer =
858 register_json_header<ThisMethod, T &, ReturnType &, Args...>(
859 arg_names, method_name);
860 initializer.json_serializer["node_type"] = "method";
861 initializer.node_type = NodeType::method;
862 const bool output_is_elementary =
863 is_registered_elementary_type<ReturnType>();
864 if (output_is_elementary)
865 set_output_only(initializer, 1);
866
867 // Add to the initializer the emtpy executor.
868 initializer.executor =
869 [ptr, &initializer, method_name, output_is_elementary](
870 const NodeObjectPtr &, std::vector<NodeObjectPtr> args)
871 -> std::shared_ptr<entt::meta_any> {
872 if (args.size() != 2 + sizeof...(Args))
873 throw std::runtime_error("Wrong number of arguments.");
874 auto &obj = args[0]->get<T>();
875 if (output_is_elementary && !args[1]->ready())
876 (*args[1])();
877 auto &ret = args[1]->get<ReturnType>();
878
879 // Create a copy of args without the first two elements for function
880 // execution
881 std::vector<NodeObjectPtr> function_args(args.begin() + 2,
882 args.end());
883
884 slog_debug("%s: %s",
885 method_name.c_str(),
886 initializer.type_name.c_str());
887 auto tuple = detail::cast_args<Args...>(function_args);
888 ret = std::apply(
889 [&obj, ptr](auto &&...unpackedArgs) {
890 return (obj.*ptr)(*unpackedArgs...);
891 },
892 tuple);
893 return std::make_shared<entt::meta_any>(
894 std::make_shared<ThisMethod>(ptr));
895 };
896 }

1421 template <typename T, typename... Args>
1422 inline auto
1423 NodeObject::register_json_header(const std::vector<std::string> &arg_names,
1424 const std::string &suffix)
1425 -> detail::NodeObjectInitializer &
1426 {
1427 auto &initializer = register_json_header<T>(suffix);
1428
1429 // Now take care of the arguments.
1430 std::vector<NodeObjectPtr> args = {std::make_shared<NodeObject>(
1431 std::shared_ptr<std::remove_cv_t<std::remove_reference_t<Args>>>())...};
1432
1433 std::vector<ConnectionType> arg_connection_types = {
1434 connection_type<Args>()...};
1435
1436 if (args.size() != arg_names.size())
1437 throw std::runtime_error("Wrong number of arguments.");
1438
1439 for (unsigned int i = 0; i < args.size(); ++i)
1440 {
1441 initializer.json_serializer["arguments"][i]["name"] = arg_names[i];
1442 initializer.json_serializer["arguments"][i]["type"] = args[i]->hash();
1443 initializer.json_serializer["arguments"][i]["connection_type"] =
1444 magic_enum::enum_name(arg_connection_types[i]);
1445 if ((arg_connection_types[i] & ConnectionType::input) !=
1446 ConnectionType::none)
1447 initializer.json_serializer["inputs"].push_back(i);
1448 if ((arg_connection_types[i] & ConnectionType::output) !=
1449 ConnectionType::none)
1450 initializer.json_serializer["outputs"].push_back(i);
1451 }
1452 return initializer;
1453 }

1560 template <typename B, typename T, typename... Args>
1561 inline auto
1562 NodeObject::register_derived_type(const std::vector<std::string> &arg_names)
1563 -> detail::NodeObjectInitializer &
1564 {
1565 auto &initializer = register_type<T, Args...>(arg_names);
1566 auto &base_initializer = register_abstract_type<B>();
1567
1568 base_initializer.json_serializer["derived"].push_back(
1569 initializer.json_serializer["type"]);
1570
1571 initializer.json_serializer["base"] =
1572 base_initializer.json_serializer["type"];
1573
1574 // Register entt conversion shared_ptr<Derived> -> shared_ptr<Base>
1575 using stored_derived =
1576 std::shared_ptr<std::remove_cv_t<std::remove_reference_t<T>>>;
1577 entt::meta_factory<stored_derived>()
1578 .template conv<&detail::shared_ptr_to_base<B, T>>();
1579
1580 // Add the conversion to the base class.
1581 initializer.to_base =
1582 [](std::shared_ptr<entt::meta_any> a) -> std::shared_ptr<entt::meta_any> {
1583 const auto ptr = a->template try_cast<std::shared_ptr<T>>();
1584 if (ptr == nullptr)
1585 throw std::runtime_error("Could not cast derived type to base.");
1586 return std::make_shared<entt::meta_any>(
1587 std::static_pointer_cast<B>(*ptr));
1588 };
1589 return initializer;

1627 NodeObject::get_shared() -> std::shared_ptr<T>
1628 {

1632 std::shared_ptr<type> ptr;

1643 if (!(new_object && *new_object))
1644 throw std::runtime_error("New object does not have value.");
1645 if (!(detail::hash(new_object) == j.at("base").get<std::string>()))
1646 throw std::runtime_error("New object does not have the right hash.");
1647 const auto cast_ptr =
1648 new_object->template try_cast<std::shared_ptr<type>>();
1649 if (cast_ptr == nullptr)
1650 throw std::runtime_error("Could not cast converted object to " +
1651 std::string(boost::core::type_name<type>()));
1652 ptr = *cast_ptr;
1653 }
1654 else
1655 {
1656 const auto cast_ptr =
1657 object->template try_cast<std::shared_ptr<type>>();
1658 if (cast_ptr == nullptr)
1659 throw std::runtime_error(
1660 "Could not cast object to shared pointer of type " +
1661 std::string(boost::core::type_name<type>()) +
1662 " from object of type " + boost::core::demangle([&]() {
1663 const char *named = object->type().name();
