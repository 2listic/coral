89 template <typename T>
90 inline auto
91 hash(const std::string &suffix = "") -> std::string
92 {
93 using underlying_type = std::remove_cv_t<std::remove_reference_t<T>>;
94 using stored_type = std::shared_ptr<underlying_type>;
95
96 const auto alias_it =
97 detail::type_aliases().find(typeid(underlying_type).hash_code());
98
99 const auto resolved_underlying = entt::resolve<underlying_type>();
100 std::string base_identifier =
101 (alias_it != detail::type_aliases().end()) ?
102 alias_it->second :
103 (resolved_underlying ?
104 detail::type_identifier(resolved_underlying) :
105 std::string(entt::type_id<underlying_type>().name()));
106
107 const char *base_name_ptr = detail::store_identifier(base_identifier);
108 const auto type_id = entt::hashed_string{base_name_ptr}.value();
109 entt::meta_factory<stored_type>().type(type_id, base_name_ptr);
110
111 if (suffix.empty())
112 return base_identifier;
113
114 // For function/method types, prefer the supplied suffix (usually a name)
115 // over the full signature to keep identifiers readable and stable.
116 if (base_identifier.find('(') != std::string::npos)
117 {
118 detail::store_identifier(suffix);
119 return suffix;
120 }
121
122 const auto identifier = base_identifier + suffix;
123 detail::store_identifier(identifier);
124 return identifier;
125 }

552 template <typename T, typename... Args>
553 static auto
554 register_json_header(const std::vector<std::string> &arg_names,
555 const std::string &suffix = "")
556 -> NodeObjectInitializer &
557 {
558 auto &initializer = register_json_header<T>(suffix);
559
560 // Now take care of the arguments.
561 std::vector<std::shared_ptr<NodeObject>> args = {std::make_shared<
562 NodeObject>(
563 std::shared_ptr<std::remove_cv_t<std::remove_reference_t<Args>>>())...};
564
565 std::vector<ConnectionType> arg_connection_types = {
566 connection_type<Args>()...};
567
568 if (args.size() != arg_names.size())
569 throw std::runtime_error("Wrong number of arguments.");
570
571 for (unsigned int i = 0; i < args.size(); ++i)
572 {
573 initializer.json_serializer["arguments"][i]["name"] = arg_names[i];
574 initializer.json_serializer["arguments"][i]["type"] = args[i]->hash();
575 initializer.json_serializer["arguments"][i]["connection_type"] =
576 magic_enum::enum_name(arg_connection_types[i]);
577 if ((arg_connection_types[i] & ConnectionType::input) !=
578 ConnectionType::none)
579 initializer.json_serializer["inputs"].push_back(i);
580 if ((arg_connection_types[i] & ConnectionType::output) !=
581 ConnectionType::none)
582 initializer.json_serializer["outputs"].push_back(i);
583 }
584 return initializer;
