75 static uint16_t
76 parse_slog_flags(const std::string &spec)
77 {
78 const auto lowered = to_lower_copy(spec);
79 if (lowered.empty() || lowered == "all")
80 return SLOG_FLAGS_ALL;
81 if (lowered == "none" || lowered == "0")
82 return 0;
83
84 uint16_t flags = 0;
85 std::string token;
86 std::istringstream ss(lowered);
87 while (std::getline(ss, token, ','))
88 {
89 token.erase(0, token.find_first_not_of(" \t\n\r"));
90 token.erase(token.find_last_not_of(" \t\n\r") + 1);
91 if (token.empty())
92 continue;
93
94 if (token == "all")
95 {
96 flags |= SLOG_FLAGS_ALL;
97 continue;
98 }
99 if (token == "none" || token == "0")
100 continue;
101 if (token == "warning")
102 token = "warn";
103
104 std::string enum_name = token;
105 if (enum_name.rfind("slog_", 0) != 0 && enum_name.rfind("SLOG_", 0) != 0)
106 enum_name = "SLOG_" + to_lower_copy(token);
107
108 // magic_enum expects the exact enumerator name, but supports case-insensitive matching.
109 auto value = magic_enum::enum_cast<slog_flag_t>(enum_name, magic_enum::case_insensitive);
110 if (!value)
111 throw std::runtime_error("Unknown slog flag: '" + token + "'");
112
113 flags |= static_cast<uint16_t>(*value);
114 }
115 return flags;
116 }
